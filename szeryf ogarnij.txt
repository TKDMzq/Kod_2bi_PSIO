{

    for (int i=0;i<20;i++)
    {
        *wsk=rand()%8;

    }
    for(i=0; i<20; i++)

        *wsk1=rand()%8;
        wsk1=wsk1+1;
    }

    for(i=0;i<20;i++)
    {
        cout <<*wsk<<" ";
        wsk=wsk+1;
    }
    cout<<endl;
    for(i=0; i<20; i++)
    {
        cout<<*wsk1<<" ";
        wsk1=wsk1+1;
    }

    wsk1=pat;
    for(i=0; i<20; i++)
    {
        if(*wsk==*wsk1)

         wsk1=wsk1+1;

    }
